021    package com.colorfulsoftware.atom;
022    
023    import java.io.Serializable;
024    
025    /**
026     * This class represents an Atom 1.0 attribute.
027     * 
028     * @see <a
029     *      href="http://www.atomenabled.org/developers/syndication/atom-format-spec.php">Atom
030     *      Syndication Format</a>
031     * @author Bill Brown
032     * 
033     *         <pre>
034     *      atomCommonAttributes =
035     *          attribute xml:base { atomUri }?,
036     *          attribute xml:lang { atomLanguageTag }?,
037     *          undefinedAttribute*
038     * </pre>
039     */
040    public class Attribute implements Serializable {
041    
042            private static final long serialVersionUID = 5289914532308394264L;
043            private final String name;
044            private final String value;
045    
046            // use the factory method in the FeedDoc.
047            Attribute(String name, String value) throws AtomSpecException {
048                    // specification customization
049                    if (name == null || name.equals("")) {
050                            throw new AtomSpecException("Attribute names SHOULD NOT be blank.");
051                    }
052                    this.name = name;
053                    this.value = (value == null) ? "" : value;
054            }
055    
056            Attribute(Attribute attribute) {
057                    this.name = attribute.name;
058                    this.value = attribute.value;
059            }
060    
061            /**
062             * 
063             * @return the name of this attribute
064             */
065            public String getName() {
066                    return name;
067            }
068    
069            /**
070             * 
071             * @return the value of this attribute
072             */
073            public String getValue() {
074                    return value;
075            }
076    
077            /**
078             * Shows the contents of the element's attribute in the form of '
079             * attrName="attrValue"'.
080             */
081            @Override
082            public String toString() {
083                    return " " + name + "=\"" + value + "\"";
084            }
085            
086            @Override
087            public boolean equals(Object obj) {
088                    if (obj == this) {
089                            return true;
090                    }
091                    if (!(obj instanceof Attribute)) {
092                            return false;
093                    }
094                    return this.toString().equals(obj.toString());
095            }
096            
097            @Override public int hashCode() {
098                    return toString().hashCode();
099            }
100    }
